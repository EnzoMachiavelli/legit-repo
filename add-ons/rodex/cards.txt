automacro RodexCards {
	Zeny >= 2500
	exclusive 1
	macro_delay 2
	timeout 1800
	call MailCards
}

macro MailCards {
	$cards = &InventoryType(card)
	@cardList = &split(',', $cards)
	
	if ($cardList[0] == -1) call noItemsAvailable
	if (&config (rodexReceiver) == $.name) call dontSend "rodexCard"
	if (&config (rodexReceiver) != $.name) goto sendIt
	
	:sendIt
		
		# set details
		$receiver = &config (rodexReceiver)
		$mailTitle = TITLE
		$mailBody = TITLE
		
		# open rodex
		do rodex open
		do rodex write $receiver
		do rodex settarget $receiver 
		do rodex settitle $mailTitle
		do rodex setbody $mailBody
		do rodex setzeny 0
		
		if (@cardList > 5) goto max
		if (@cardList < 5) goto min
		
		:max
			$i = 0
			$maxItem = 5				
			
			$i = 0;
			while ($i < $maxItem) {
				log Putting Card with Index: $cardList[$i]
				do rodex add $cardList[$i] 1
				$i++
			}
			
			do rodex send
			do rodex close
			stop
		:min
			$i = 0
			$maxItem = @cardList				
			
			$i = 0;
			while ($i < $maxItem) {
				log Putting Card with Index: $cardList[$i]
				do rodex add $cardList[$i] 1
				$i++
			}
			
			do rodex send
			do rodex close
			stop
}
